// Hexadecimal and Base64 Obfuscation Example

// Hexadecimal obfuscation
var hexString = "\\x68\\x65\\x6c\\x6c\\x6f"; // Represents "hello"

// Base64 obfuscation
var base64String = "aGVsbG8="; // Represents "hello"

function decodeHex(hex) {
    return decodeURIComponent(hex.replace(/\\x/g, '%'));
}

function decodeBase64(base64) {
    return atob(base64);
}

console.log(decodeHex(hexString)); // Output: hello
console.log(decodeBase64(base64String)); // Output: hello
